Last updated at: May 27, 2018

--result-file=SQL_FILE_PATH \
--complete-insert SCHEMA_NAME [TABLE_NAME] \
-uYOUR_USERNAME \
-pYOUR_PASSWORD \
--host=HOST \
--port=PORT
